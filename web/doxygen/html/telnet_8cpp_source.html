00240         const ConnectionInfo cli_Info = m_tkConnections.GetAt(it);

00246     timeval t_Timeout, *pt_Timeout = NULL;
00247     if (I_Milli >= 0)
00248     {

00250         t_Timeout.tv_usec = (I_Milli * 1000) % 1000000;
00251         pt_Timeout = & t_Timeout;

00253 
00254     // Check the server is still running.
00255     if (m_iServerSocket < 0)
00256     {
00257         return true;

00259

00271     {
00272         return true;
