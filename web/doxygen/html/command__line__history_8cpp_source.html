00052 const bool CmdLineHistory::Push(const CmdLineEdition& CLI_Line)
00053 {
00054     if (// Check the line is not empty.
00055         (! CLI_Line.GetLine().IsEmpty())
00056         // Check it is not the same as the previous one.
00057         && ((m_qHistory.GetCount() <= 1) || (CLI_Line.GetLine() != GetLine(1).GetLine())))
00058     {
00059         // Limit to HISTORY_STACK_SIZE.
00060         while (m_qHistory.GetCount() > m_uiStackSize)
00061         {
00062             m_qHistory.RemoveTail();
00063         }
00064 
00065         // Ensure there is at least one element.
00066         if (m_qHistory.IsEmpty())
00067         {
00068             if (! m_qHistory.AddTail(CmdLineEdition()))
00069             {
00070                 return false;
00071             }
00072         }
00073 
00074         // Backup the current line.
00075         m_qHistory.GetHead() = CLI_Line;
00076 
00077         // Add a new empty element.
00078         m_qHistory.AddHead(CmdLineEdition());
00079     }
00080 
00081     m_uiNavigationIndex = 0;
00082     m_bNavigation = false;
00083 
00084     return true;
00085 }
