00588 void Shell::OnHelp(const bool B_PromptMenu, const bool B_Completion)
00589 {

00593         CommandLine cli_CommandLine;
00594         if (cli_CommandLine.Parse(*m_qMenus.GetTail(), m_cliCmdLine.GetLeft(), false))
00595         {
00596             // Research matching elements.
00597             Element::List cli_Exact(MAX_WORDS_PER_NODE), cli_Elements(MAX_WORDS_PER_NODE);
00598             if (! cli_CommandLine.GetLastElement().FindElements(
00599                     cli_Exact, cli_Elements, cli_CommandLine.GetLastWord()))
00600             {
00601                 return;
00602             }
00603 
00604             // Initialize completion so far.
00605             tk::String str_CompletionSoFar(MAX_WORD_LENGTH);
00606             // Determine whether help should be printed, and completion activated.
00607             bool b_PrintHelpList = false;
00608 
00609             if (! B_Completion)
00610             {
00611                 // Just print help.
00612                 b_PrintHelpList = true;
00613             }
00614             else
00615             {
00616                 // Print help and do completion depending on the possible nodes.
00617                 if (cli_Elements.IsEmpty())
00618                 {
00619                     // No matching element.
00620                     Beep();
00621                 }
00622                 else if ((cli_Elements.GetCount() == 1)
00623                         && (! dynamic_cast<const Endl*>(cli_Elements.GetHead()))
00624                         && (! dynamic_cast<const Param*>(cli_Elements.GetHead())))
00625                 {
00626                     // Straight forward completion.
00627                     m_cliCmdLine.Delete(GetStream(ECHO_STREAM), - cli_CommandLine.GetNumBackspacesForCompletion());
00628                     m_cliCmdLine.Put(GetStream(ECHO_STREAM), cli_Elements.GetHead()->GetKeyword());
00629                     m_cliCmdLine.Put(GetStream(ECHO_STREAM), ' ');
00630                 }
00631                 else
00632                 {
00633                     // Look for completion so far.
00634                     bool b_InitCompletionSoFar = true;
00635                     for (   Element::List::Iterator it = cli_Elements.GetIterator();
00636                             cli_Elements.IsValid(it);
00637                             cli_Elements.MoveNext(it))
00638                     {
00639                         if (dynamic_cast<const Param*>(cli_Elements.GetAt(it))
00640                             || dynamic_cast<const Endl*>(cli_Elements.GetAt(it)))
00641                         {
00642                             // No completion for parameters and end of lines.
00643                             str_CompletionSoFar.Set("");
00644                             break;
00645                         }
00646                         else if (const Element* const pcli_Element = dynamic_cast<const Element*>(cli_Elements.GetAt(it)))
00647                         {
