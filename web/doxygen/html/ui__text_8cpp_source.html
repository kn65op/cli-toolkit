00268         {
00269             for (   unsigned int ui_CharCount = 0;

00275                 if (it.m_uiBottomChar >= tk_Line.GetLength())
00276                 {
00277                     // End of line.
00278                     CLI_ASSERT(it.m_cliScreenInfo.GetSafeWidth() > 0);
00279                     if ((! it.m_cliScreenInfo.GetbWrapLines())
00280                         // When line wrapping is on, a line ending at the right end of the screen should not cause an empty line below.
00281                         // Thus, print a carriage return only in the following cases:
00282                         || (it.m_uiBottomChar == 0)                                         // Regular empty line.
00283                         || (it.m_cliScreenInfo.GetSafeWidth() <= 0)                         // (Avoid invalid modulo, should never occure however)
00284                         || (it.m_uiBottomChar % it.m_cliScreenInfo.GetSafeWidth() != 0))    // Line not ending at the right end the screen.
00285                     {
00286                         CLI_Out << endl;
00287                     }

00289                     it.m_uiBottomChar = 0;

00292                 else if (ui_CharCount >= it.m_cliScreenInfo.GetSafeWidth())
00293                 {
00294                     // Out of line.
00295                     if (! it.m_cliScreenInfo.GetbWrapLines())

00297                         CLI_Out << endl;
