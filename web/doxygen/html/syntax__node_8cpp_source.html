00125 const bool SyntaxNode::FindElements(
00126         Element::List& CLI_ExactList,
00127         Element::List& CLI_NearList,
00128         const char* const STR_Keyword
00129         ) const
00130 {
00131     // For each child...
00132     for (   Element::List::Iterator it = m_cliElements.GetIterator();
00133             m_cliElements.IsValid(it);
00134             m_cliElements.MoveNext(it))
00135     {
00136         if (const Element* const pcli_Element = m_cliElements.GetAt(it))
00137         {
00138             if (0) {}
00139             else if (const SyntaxTag* const pcli_Tag = dynamic_cast<const SyntaxTag*>(pcli_Element))
00140             {
00141                 // Propagate call over child non hollow tag.
00142                 if (! pcli_Tag->GetbHollow())
00143                 {
00144                     if (! pcli_Tag->FindElements(CLI_ExactList, CLI_NearList, STR_Keyword))
00145                     {
00146                         return false;
00147                     }
00148                 }
00149             }
00150             else if (const SyntaxRef* const pcli_Ref = dynamic_cast<const SyntaxRef*>(pcli_Element))
00151             {
00152                 // Propagate call over referenced tag.
00153                 if (! pcli_Ref->GetTag().FindElements(CLI_ExactList, CLI_NearList, STR_Keyword))
00154                 {
00155                     return false;
00156                 }
00157             }
00158             else if (STR_Keyword == NULL)
00159             {
00160                 // No keyword begun.
00161                 // Retrieve all sub-elements.
00162                 if (! CLI_NearList.AddTail(pcli_Element))
00163                 {
00164                     GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
00165                 }
00166             }
00167             else
00168             {
00169                 // A beginning of word has been given.
00170                 const tk::String str_Keyword(MAX_WORD_LENGTH, STR_Keyword);
00171 
00172                 if (const Param* const pcli_Param = dynamic_cast<const Param*>(pcli_Element))
00173                 {
00174                     // If the child element is a parameter, check SetstrValue() works for it.
00175                     if (str_Keyword != "\n")
00176                     {
00177                         if (pcli_Param->SetstrValue(str_Keyword))
00178                         {
00179                             if (! CLI_NearList.AddTail(pcli_Param))
00180                             {
00181                                 GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
00182                             }
00183                             if (! CLI_ExactList.AddTail(pcli_Param))
00184                             {
00185                                 GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
00186                             }
00187                         }
00188                     }
00189                 }
00190                 else if (pcli_Element->GetKeyword().SubString(0, str_Keyword.GetLength()) == str_Keyword)
00191                 {
00192                     // Check the beginning of the word for other elements.
00193                     if (! CLI_NearList.AddTail(pcli_Element))
00194                     {
00195                         GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
00196                     }
00197                     if (str_Keyword.GetLength() == pcli_Element->GetKeyword().GetLength())
00198                     {
00199                         if (! CLI_ExactList.AddTail(pcli_Element))
00200                         {
00201                             GetTraces().Trace(INTERNAL_ERROR) << "SyntaxNode::FindElements(): Not enough space in CLI_ExactList." << endl;
00202                         }
00203                     }
00204                 }
00205             }
00206         }
00207     }
