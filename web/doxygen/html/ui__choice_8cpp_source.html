00041         static const ResourceString Choice2Str(
00042                 const int I_Choice,                             
00043                 const tk::Queue<ResourceString>& TK_Choices     
00044                 )
00045         {
00046             // Loop over the choice list until the number of elements to skip has been reached.
00047             tk::Queue<ResourceString>::Iterator it =  TK_Choices.GetIterator();
00048             for (   int i_ElementsToSkip = I_Choice;
00049                     (i_ElementsToSkip >= 0) && TK_Choices.IsValid(it);
00050                     i_ElementsToSkip --)
00051             {
00052                 if (i_ElementsToSkip == 0)
00053                 {
00054                     return TK_Choices.GetAt(it);
00055                 }
00056                 else
00057                 {
00058                     TK_Choices.MoveNext(it);
00059                 }
00060             }
00061 
00062             // Default to an empty string.
00063             return ResourceString();
00064         }

00085         const int Choice::GetChoice(void) const
00086         {
00087             const tk::String str_Line = Line::GetLine();
00088             int i_Choice = 0;
00089             tk::Queue<int> tk_Near(10);
00090             for (   tk::Queue<ResourceString>::Iterator it = m_tkChoices.GetIterator();
00091                     m_tkChoices.IsValid(it);
00092                     m_tkChoices.MoveNext(it), i_Choice++)
00093             {
00094                 const tk::String str_Choice = m_tkChoices.GetAt(it).GetString(m_eLang);
00095 
00096                 // Check whether the line matches the choice exactly.
00097                 if (str_Line == str_Choice)
00098                 {
00099                     return i_Choice;
00100                 }
00101 
00102                 // Check whether the line starts like the choice (case insensitive).
00103                 if (str_Line.ToUpper() == str_Choice.SubString(0, str_Line.GetLength()).ToUpper())
00104                 {
00105                     tk_Near.AddTail(i_Choice);
00106                 }
00107             }
00108 
00109             // If only one near result, then this one is the choice.
00110             if (tk_Near.GetCount() == 1)
00111             {
00112                 return tk_Near.GetHead();
00113             }
00114 
00115             // No matching choice.
00116             return -1;
00117         }

00178         void Choice::MoveChoice(const int I_Way, const unsigned int UI_Increment)
00179         {
00180             const int i_Increment = I_Way * (int) ((UI_Increment == 0) ? 1 : UI_Increment);
00181             const int i_Choice = GetChoice();
00182 
00183             if (i_Choice < 0)
00184             {
00185                 // Undefined choice: restore default.
00186                 Beep();
00187                 ResetToDefault();
00188             }
00189             else if (i_Choice + i_Increment < 0)
00190             {
00191                 // Top of choice list.
00192                 // Possibly beep.
00193                 if (i_Choice <= 0)
00194                 {
00195                     Beep();
00196                 }
00197                 // Reprint.
00198                 Line::SetLine(Choice2Str(0, m_tkChoices).GetString(m_eLang), false, true);
00199             }
00200             else if (i_Choice + i_Increment >= (int) m_tkChoices.GetCount())
00201             {
00202                 // Bottom of choice list.
00203                 // Possibly beep.
00204                 if (i_Choice >= ((int) m_tkChoices.GetCount()) - 1)
00205                 {
00206                     Beep();
00207                 }
00208                 // Reprint.
00209                 Line::SetLine(Choice2Str(m_tkChoices.GetCount() - 1, m_tkChoices).GetString(m_eLang), false, true);
00210             }
00211             else
00212             {
00213                 // Move to next element.
00214                 Line::SetLine(Choice2Str(GetChoice() + i_Increment, m_tkChoices).GetString(m_eLang), false, true);
00215             }
00216         }
