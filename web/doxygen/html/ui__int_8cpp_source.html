00114                     // Completely out of bounds: return to max value.

00119                 {

00124                     // Lower bound already reached: ensure min value.

00129             case PAGE_UP:
00130                 if (GetInt() >= m_iMaxValue)

00132                     // Upper bound already reached.
00133                     Beep();

00135                 // Ensure max value.
00136                 Line::SetLine(Int2Str(m_iMaxValue), false, false);
