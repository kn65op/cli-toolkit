00090         void Less::OnKey(const KEY E_KeyCode)
00091         {
00092             // Ensure m_puiTextIt is valid.
00093             CLI_ASSERT(m_puiTextIt != NULL);
00094             if (m_puiTextIt == NULL) { Quit(); }
00095             else {
00096                 switch (E_KeyCode)
00097                 {
00098                 case KEY_BEGIN:
00099                     m_uiText.Begin(*m_puiTextIt);
00100                     PrintScreen();
00101                     break;
00102                 case PAGE_UP:
00103                     if (m_uiText.PageUp(*m_puiTextIt)) PrintScreen();
00104                     else Beep();
00105                     break;
00106                 case KEY_UP:
00107                     if (m_uiText.LineUp(*m_puiTextIt)) PrintScreen();
00108                     else Beep();
00109                     break;
00110                 case KEY_DOWN:
00111                 case ENTER:
00112                     if (m_uiText.LineDown(*m_puiTextIt, NULL)) PrintScreen();
00113                     else Beep();
00114                     break;
00115                 case PAGE_DOWN:
00116                 case SPACE:
00117                     if (m_uiText.PageDown(*m_puiTextIt, NULL)) PrintScreen();
00118                     else Beep();
00119                     break;
00120                 case KEY_END:
00121                     m_uiText.End(*m_puiTextIt, NULL);
00122                     PrintScreen();
00123                     break;
00124                 case KEY_q:
00125                 case KEY_Q:
00126                 case ESCAPE:
00127                 case BREAK:
00128                 case LOGOUT:
00129                 case NULL_KEY:
00130                     // Stop display
00131                     Quit();
00132                     break;
00133                 default:
00134                     // Non managed character: beep.
00135                     Beep();
00136                     break;
00137                 }
00138             }
00139         }

00141         void Less::PrintScreen(void)
00142         {
00143             const OutputDevice::ScreenInfo cli_ScreenInfo = GetStream(OUTPUT_STREAM).GetScreenInfo();
00144 
00145             // Then let current bottom position move one page down while printing the page.
00146             const StringDevice cli_Out(cli_ScreenInfo.GetSafeHeight() * (cli_ScreenInfo.GetSafeWidth() + 1), false);
00147             CLI_ASSERT(m_puiTextIt != NULL);
00148             if (m_puiTextIt != NULL) { m_uiText.PrintPage(*m_puiTextIt, cli_Out, true); }
00149 
00150             // Eventually clean out the screen and print out the computed string.
00151             m_cliLessLine.Reset();
00152             GetStream(OUTPUT_STREAM).CleanScreen();
00153             GetStream(OUTPUT_STREAM) << cli_Out.GetString();
00154             m_cliLessLine.Put(GetStream(OUTPUT_STREAM), tk::String(10, ":"));
00155         }
